    /* ---- shadcn/ui RTL patches (D-01) ---- */
    /* Radix Popover/Dropdown use Floating UI which positions correctly under dir="rtl",
       BUT shadcn's default animations and a few alignment utilities assume LTR.
       Override only the broken bits — no fork, no extra deps. */

    [dir="rtl"] [data-radix-popper-content-wrapper] {
      /* Floating UI honors dir, but ensure transform-origin is mirrored for animations */
      transform-origin: var(--radix-popper-transform-origin);
    }

    /* Popover + DropdownMenu: flip side-offset animation direction in RTL.
       shadcn slide-in-from-left-2 / slide-in-from-right-2 hardcode LTR. */
    [dir="rtl"] [data-state="open"][data-side="left"] {
      animation-name: slideInFromRight;
    }
    [dir="rtl"] [data-state="open"][data-side="right"] {
      animation-name: slideInFromLeft;
    }

    @keyframes slideInFromRight {
      from { opacity: 0; transform: translateX(8px); }
      to { opacity: 1; transform: translateX(0); }
    }
    @keyframes slideInFromLeft {
      from { opacity: 0; transform: translateX(-8px); }
      to { opacity: 1; transform: translateX(0); }
    }

    /* react-day-picker (Calendar) — caption nav buttons are positioned absolutely
       with left/right; mirror them under RTL. */
    [dir="rtl"] .rdp-nav_button_previous {
      left: auto;
      right: 0.25rem;
      transform: scaleX(-1); /* flip chevron icon */
    }
    [dir="rtl"] .rdp-nav_button_next {
      right: auto;
      left: 0.25rem;

      transform: scaleX(-1);
    }
