    "use client";
    import { useRouter } from "next/navigation";
    import { useState, useTransition } from "react";
    import Link from "next/link";
    import { useTranslations } from "next-intl";
    import { Button } from "@/components/ui/button";

    interface EventRow {
      id: string;
      title: string;
      status: "draft" | "pending";
      updatedAt: Date;
      startAt: Date;
    }
    export function EventList({ events }: { events: EventRow[] }) {
      const t = useTranslations("dashboard");
      const router = useRouter();
      const [, startTransition] = useTransition();
      const [errorId, setErrorId] = useState<string | null>(null);

      async function handleDelete(id: string) {

          setErrorId(id);

        return (

        <div>
          <div className="mb-4">
            <Link href="/events/new"><Button>{t("newEvent")}</Button></Link>
          </div>
          <ul className="space-y-2">
            {events.map((e) => (
              <li key={e.id} className="border rounded p-3 flex items-center gap-3">
                <span className="font-medium">{e.title || t("untitled")}</span>
                <span className="text-xs px-2 py-0.5 rounded bg-gray-100">{t(`status.${e.status}`)}</span>
                <span className="ms-auto flex gap-2">
                  <Link href={`/events/${e.id}/edit?step=1`} className="underline">
                    {e.status === "draft" ? t("resume") : t("view")}
                  </Link>
                  {e.status === "draft" && (
                    <Button variant="ghost" onClick={() => handleDelete(e.id)}>
                      {t("delete")}
                    </Button>
                  )}
                </span>
                {errorId === e.id && <span className="text-red-500 text-xs">{t("deleteError")}</span>}
              </li>
            ))}
          </ul>
        </div>
      );
    }

    Create `components/wizard/WizardShell.tsx`:
    ```typescript
    "use client";
    import { useRouter, useSearchParams } from "next/navigation";
    import { useState } from "react";
    import { useTranslations } from "next-intl";
    import type { EventDraftViewModel, AcademicYearView, EventTypeOption } from "./types";
    import { WizardStep1 } from "./WizardStep1";
    import { WizardStep2 } from "./WizardStep2";
    import { WizardStep3 } from "./WizardStep3";
    import { WizardStep4 } from "./WizardStep4";
    import { WizardStep5 } from "./WizardStep5";
    import { WizardStep6 } from "./WizardStep6";
    import { WizardStep7 } from "./WizardStep7";
    import { Button } from "@/components/ui/button";

    interface Props {
      initial: EventDraftViewModel;
      allowedGrades: number[];
      academicYear: AcademicYearView;
      eventTypes: EventTypeOption[];
    }

    export function WizardShell({ initial, allowedGrades, academicYear, eventTypes }: Props) {
      const t = useTranslations("wizard");
      const router = useRouter();
      const params = useSearchParams();
      const step = Math.min(Math.max(Number(params.get("step") ?? "1"), 1), 7);
      const [data, setData] = useState<EventDraftViewModel>(initial);
      const [saving, setSaving] = useState(false);
      const [conflict, setConflict] = useState(false);
      const [error, setError] = useState<string | null>(null);

      async function patch(patchBody: Record<string, unknown>): Promise<boolean> {
        setSaving(true);
        setError(null);
        try {
          const res = await fetch(`/api/v1/events/${data.id}`, {
            method: "PATCH",
            headers: {
              "Content-Type": "application/json",
              "If-Match": String(data.version),
            },
            body: JSON.stringify(patchBody),
          });
          if (res.status === 409) {
            setConflict(true);
            return false;
          }
          if (!res.ok) {
            const body = await res.json().catch(() => ({}));
            setError(body.error ?? "error");
            return false;
          }
          const body = await res.json();
          setData((prev) => ({ ...prev, ...patchBody, version: body.version }));
          return true;
        } finally {
          setSaving(false);
        }
      }

      async function handleNext(stepData: Record<string, unknown>) {
        // Step 6 (review) returns {} from stepPayload — don't PATCH with empty body (avoids needless version bump)
        if (Object.keys(stepData).length === 0) {
          router.push(`?step=${step + 1}`);
          return;
        }
        const ok = await patch(stepData);
        if (ok) router.push(`?step=${step + 1}`);
      }
      async function handlePrev() {
        if (step > 1) router.push(`?step=${step - 1}`);
      }
      async function handleSubmit() {
        const res = await fetch(`/api/v1/events/${data.id}/submit`, { method: "POST" });
        if (res.ok) router.push("/dashboard");
        else {
          const body = await res.json().catch(() => ({}));
          setError(body.error ?? "submit_failed");
        }
      }

      return (
        <main className="max-w-2xl mx-auto p-6">
          <div className="mb-4 text-sm text-gray-600">{t("step", { n: step })}</div>
          {conflict && (
            <div role="alert" className="mb-4 p-3 bg-red-50 border border-red-300 rounded">
              {t("conflictToast")}
            </div>
          )}
          {error && !conflict && (
            <div role="alert" className="mb-4 p-3 bg-yellow-50 border border-yellow-300 rounded">
              {error}
            </div>
          )}
          {step === 1 && <WizardStep1 data={data} eventTypes={eventTypes} onChange={(d) => setData((p) => ({ ...p, ...d }))} />}

          {step === 3 && <WizardStep3 data={data} allowedGrades={allowedGrades} onChange={(d) => setData((p) => ({ ...p, ...d }))} />}
          {step === 4 && <WizardStep4 data={data} academicYear={academicYear} onChange={(d) => setData((p) => ({ ...p, ...d }))} />}
          {step === 5 && <WizardStep5 data={data} onChange={(d) => setData((p) => ({ ...p, ...d }))} />}
          {step === 6 && <WizardStep6 data={data} eventTypes={eventTypes} />}
          {step === 7 && <WizardStep7 data={data} onSubmit={handleSubmit} />}

          <div className="mt-6 flex gap-2 justify-between">
            <Button variant="ghost" disabled={step === 1 || saving} onClick={handlePrev}>{t("prev")}</Button>
            {step < 7 ? (
              <Button
                disabled={saving || conflict}
                onClick={() => handleNext(stepPayload(step, data))}
              >
                {saving ? t("saving") : t("next")}
              </Button>
            ) : null}
          </div>
        </main>
      );
    }

    function stepPayload(step: number, data: EventDraftViewModel): Record<string, unknown> {
      switch (step) {
        case 1: return { eventTypeId: data.eventTypeId };
        case 2: return { title: data.title, description: data.description };
        case 3: return { grades: data.grades };
        case 4: return { startAt: data.startAt, endAt: data.endAt, allDay: data.allDay };
        case 5: return { location: data.location };
        case 6: return {}; // review-only, no fields
        default: return {};
      }
    }
    ```
